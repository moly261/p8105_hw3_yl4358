``` r
knitr::opts_chunk$set(
  fig.width = 6,
  fig.asp = .6,
  out.width = "90%"
)
instacart_2 =
  instacart %>%
  group_by(aisle_id) %>% 
  summarize(n = n()) %>%
  arrange(desc(n)) %>% 
  filter(n >= 10000) %>% 
   ggplot(aes(x = aisle_id, y = n)) + 
    geom_point(color = "blue", size = 3) + geom_line(color = "blue") +
  labs(
    title = "The number of items ordered in each aisle",
    x = "Aisle ID",
    y = "The number of items ordered"
  ) +
  theme_bw()
instacart_2
```
